### Body

<ParamField body="file_url" type="string">
  The file URL you provide is used to identify a file that exists in your fly
  that you want to delete.
</ParamField>

```rust Rust
use reqwest::blocking::Client;
use std::collections::HashMap;

fn main() -> Result<(), reqwest::Error> {
    let client = Client::new();
    let url = "https://api.uploadfly.cloud/delete";

    let mut headers = HashMap::new();
    headers.insert("Authorization", "Bearer SECRET_KEY");

    let mut body = HashMap::new();
    body.insert("file_url", "cdn.uploadfly.cloud/xrwbua/example.png");

    let response = client
        .delete(url)
        .headers(headers)
        .json(&body)
        .send()?;

    // Handle response

    Ok(())
}
```
